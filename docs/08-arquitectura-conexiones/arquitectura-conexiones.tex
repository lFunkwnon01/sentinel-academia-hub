% ===== 1. RESUMEN =====
\section{Resumen de la Arquitectura}

Sentinel AcademIA es una plataforma \textbf{serverless multi-tenant} para
gestión de quejas universitarias con análisis automático de IA.
La arquitectura se divide en \textbf{seis zonas}:

\begin{enumerate}[leftmargin=*]
  \item \textbf{Cliente (ClientEdge)}: navegador del usuario, frontend Vue 3
  \item \textbf{AWS Lambda} (10 funciones): lógica de negocio serverless
  \item \textbf{AWS Cognito}: autenticación de usuarios
  \item \textbf{AWS S3 + DynamoDB + SQS}: almacenamiento y colas
  \item \textbf{AWS SNS + Email}: notificaciones a las autoridades
  \item \textbf{Oracle OCI}: LLM (Cohere) + Document AI (OCR)
\end{enumerate}

\subsection{Conexi\'on 1: Usuario $\rightarrow$ Frontend}

\begin{tcolorbox}[colback=blue!5, colframe=clientblue, title=\textbf{Desde $\rightarrow$ Hasta}]
\textbf{Desde:} Navegador del usuario (Chrome, Firefox, Safari) \\
\textbf{Hacia:} Frontend Vue 3 desplegado en S3 (Vercel opcional) \\
\textbf{Protocolo:} HTTPS (puerto 443)
\end{tcolorbox}

\begin{center}
\begin{tikzpicture}
  \node[clientbox, fill=blue!15] (u) at (0,0) {Usuario};
  \node[clientbox, fill=blue!15] (w) at (5,0) {Frontend Vue 3};
  \draw[arrow] (u) -- node[arrownote, above] {\textbf{HTTPS GET /}} (w);
\end{tikzpicture}
\end{center}

\textbf{Qu\'e datos fluyen:}
\begin{itemize}[leftmargin=*]
  \item \textbf{Request:} el navegador pide \texttt{https://sentinel-app.com}
  \item \textbf{Response:} S3/Vercel devuelve HTML, CSS, JS del frontend
\end{itemize}

\textbf{Configuraci\'on:}
\begin{enumerate}[leftmargin=*]
  \item Crear bucket S3: \texttt{sentinel-frontend-dev-\{account-id\}}
  \item Habilitar \textbf{Static Website Hosting}
  \item Configurar \textbf{Bucket Policy} para lectura p\'ublica
  \item Subir: \texttt{aws s3 sync dist/ s3://sentinel-frontend-dev-\{id\}/}
\end{enumerate}

\subsection{Conexi\'on 2: Usuario $\rightarrow$ API Gateway (10 Lambdas)}

\begin{tcolorbox}[colback=orange!5, colframe=awsorange, title=\textbf{Desde $\rightarrow$ Hasta}]
\textbf{Desde:} Navegador \\
\textbf{Hacia:} API Gateway $\rightarrow$ 10 AWS Lambdas \\
\textbf{Protocolo:} HTTPS REST
\end{tcolorbox}

\begin{center}
\begin{tikzpicture}
  \node[clientbox, fill=blue!15] (u) at (0,0) {Usuario};
  \node[awsbox, fill=orange!15] (l) at (5,0) {API + 10 Lambdas};
  \draw[arrow] (u) -- node[arrownote, above] {\textbf{HTTPS GET/POST}} (l);
\end{tikzpicture}
\end{center}

\textbf{Las 10 Lambdas (cada una = 1 endpoint):}
\begin{enumerate}[leftmargin=*]
  \item \texttt{health\_check} --- GET /health
  \item \texttt{create\_queja} --- POST /api/quejas
  \item \texttt{get\_queja} --- GET /api/quejas/\{id\}
  \item \texttt{list\_quejas} --- GET /api/quejas
  \item \texttt{analyze\_queja} --- Consumer de SQS
  \item \texttt{upload\_url} --- POST /api/quejas/\{id\}/upload-url
  \item \texttt{file\_processor} --- Consumer de S3
  \item \texttt{chat} --- POST /api/chat
  \item \texttt{dashboard\_metrics} --- GET /api/dashboard/metrics
  \item \texttt{notify\_email} --- Consumer de SNS
\end{enumerate}
